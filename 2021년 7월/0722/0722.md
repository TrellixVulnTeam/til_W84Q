def my_list_max2(*numbers_list):
    # print(numbers_list)
    # print(numbers_list[1])
    # print(type(numbers_list))
    max = 0
    answer = ''
    for numbers in numbers_list :
        # print(numbers)
        if sum(numbers) > max:
            max = sum(numbers)
            answer = numbers
    return answer
print(my_list_max2([10, 3], [5, 9],[11,12],[10,6]))

팩토리얼 계산 식들

```
def fact(n):
    result =1
    while n >1:
        result *= n 
        n -= 1
    return result
    
fact(5)
```

```
def factorial(n):
    if n ==1:
        return 1
    return n* factorial(n-1)
    
 factorial(5)
```

def fib_loog(n):
    if n < 2:
        return n

    result = [0, 1]
    for i in range(2, n + 1):
        result.append(result[i - 1] + result[i - 2])
    return result[-1] 
   
fib_loog(10)

class Point:
    def __init__(self,x,y):
        self.x=x
        self.y=y


class Rectangle(Point):
    def __init__(self,p1,p2):
        self.p1 = p1
        self.p2 = p2
        super().__init__(self.p1,self.p2)
    def get_area(self):
        return (p2.x -p1.x)*(p1.y -p2.y)
    def get_perimeter(self):
        return (((p2.x -p1.x)+(p1.y -p2.y))*2)
    def is_square(self):
        return ((p2.x -p1.x)==(p1.y -p2.y))

class Rectangle2(Point):
    def __init__(self, p3, p4):
        self.p3= p3
        self.p4= p4
        super().__init__(self.p3,self.p4)
    def get_area(self):
        return (p4.x - p3.x) * (p3.y- p4.y)
    def get_perimeter(self):
        return (((p4.x - p3.x) + (p3.y - p4.y)) * 2)
    def is_square(self):
        return ((p4.x - p3.x) == (p3.y - p4.y))

p1 = Point(1,3)
p2 = Point(3,1)
r1 = Rectangle(p1,p2)
print(r1.get_area())
print(r1.get_perimeter())
print(r1.is_square())
p3 = Point(3,7)
p4 = Point(6,4)
r2 = Rectangle2(p3,p4)

print(r2.get_perimeter())
